Infinite Campus Firestore Database Rules

Infinite Campus Uses Google Firebase For Storing The Articles For The News Page

Below Are The Full Infinite Campus Firestore Database Rules 


                
    rules_version = '2';
    service cloud.firestore {
        match /databases/{database}/documents {
            function isOwner() {
                return request.auth != null &&
                request.auth.uid in [
                    "jiEcu7wSifMalQxVupmQXRchA9k1"
                ];
            }
            function isTester() {
                return request.auth != null &&
                request.auth.uid in [
                    "5EQBEAbANRYAXq5qtwNgP54D65m1"
                ];
            }
            function isDev() {
                return request.auth != null &&
                request.auth.uid in [
                    "0MFrAJ3RSwayWkvHhhx34ugzOuB2"
                ];
            }
            function isAdmin() {
                return isOwner() || isTester() || isDev();
            }
            match /articles/{articleId} {
                allow read: if true;
                allow write: if isAdmin();
            }
            match /articles/{articleId}/content {
                allow read: if true;
                allow write: if isAdmin();
            }
            match /{document=**} {
                allow read, write: if isAdmin();
            }
        }
    }
